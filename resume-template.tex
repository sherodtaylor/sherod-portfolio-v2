%% Visual rules from SOUL.md:
%%   - FiraCode Mono Nerd Font everywhere (matches the website).
%%   - Single accent color used sparingly (links + heading rule).
%%   - Hairline rules between sections, no decorated headers.
%%   - No colored highlight boxes, no badges, no card grids.
%%   - Documentation-style spacing (generous line-height for mono body).

%% template; the website renders from the same YAML. SOUL.md edits update
%% both surfaces on next deploy.

}

% Skill category — title with years-of-experience tucked to the right,
% then the skill list on the next line. Matches the website's "Skills"
% section format (label + middot-separated list). #2 (description) and
% #3 ("Expert") are intentionally not rendered — both are repetitive
% bio prose, not callout info.
\newcommand{\skillcategory}[5]{%
    \textbf{\color{textdark}#1}\hfill\textcolor{textgray}{\footnotesize #4}\\%
    {\footnotesize\color{textgray} #5}%
    \vspace{0.5em}
}

\newcommand{\communityrole}[4]{%
    \textbf{\color{textdark}#1} \hfill \textcolor{textgray}{\footnotesize #4} \\
    \textcolor{textgray}{\footnotesize #2 \textperiodcentered{} #3}%
    \vspace{0.2em}
}

% Hero highlights are anti-pattern per SOUL.md ("no qualification chips").
% The generator still calls \highlightbox{}; render each as plain
% middot-separated muted text so the existing data shape works without
% emitting colored boxes.
\newcommand{\highlightbox}[3]{%
    {\footnotesize\color{textgray} #3}%
}
